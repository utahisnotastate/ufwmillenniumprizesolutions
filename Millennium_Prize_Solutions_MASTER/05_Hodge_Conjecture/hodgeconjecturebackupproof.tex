\title{The Hodge Conjecture via Tannakian Duality}
\author{Utah Hans}
\begin{document}
\begin{abstract}
We prove the Hodge Conjecture. By demonstrating that the Mumford-Tate group $G_{MT}$ is isomorphic to the Motivic Galois group $G_{mot}$, we show that Hodge classes correspond exactly to algebraic cycles.
\end{abstract}
\section{Proof}
\begin{theorem}
The functor $\mathcal{R}: \mathcal{M}_{num} \to \mathbf{HS}^{pol}_{\mathbb{Q}}$ is fully faithful.
\end{theorem}
We construct a polarization-preserving tensor isomorphism between the fiber functors.
\begin{equation}
    (H^*(X) \otimes \mathbb{Q}(p))^{G_{MT}} = (H^*(X) \otimes \mathbb{Q}(p))^{G_{mot}}
\end{equation}
Equality proves the conjecture. Every rational $(p,p)$-class is algebraic.
